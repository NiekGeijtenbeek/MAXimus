76 
77  bool yLonger=false;

80  if ( abs( shortLen )>abs( longLen )) {
81  int swap=shortLen;
82  shortLen=longLen;
83  longLen=swap;
84  yLonger=true;
85  }
86  int decInc;
87  if ( longLen==0 ) decInc=0;
88  else decInc = ( shortLen << 8 ) / longLen;
89 
90  if ( yLonger ) {
91  if ( longLen>0 ) {
92  longLen+=y;
93  for ( int j=0x80+(x<<8);y<=longLen;++y ) {

97  return;
98  }

101  w.set( coordinate( j >> 8,y ));
102  j-=decInc;
103  }
